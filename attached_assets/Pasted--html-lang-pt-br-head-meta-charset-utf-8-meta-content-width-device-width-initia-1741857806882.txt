    function updateDate() {
        const now = new Date();
        const day = String(now.getDate()).padStart(2, '0');
        const month = String(now.getMonth() + 1).padStart(2, '0');
        const year = now.getFullYear();
        const hours = String(now.getHours()).padStart(2, '0');
        const minutes = String(now.getMinutes()).padStart(2, '0');
        
        const formattedDate = `${day}/${month}/${year} ${hours}h${minutes} · Atualizado hoje`;
        
        document.getElementById('currentDate').textContent = formattedDate;
    }
    
    updateDate();
    setInterval(updateDate, 60000); // Update every minute

    // Function to balance traffic between four links
    function balanceTraffic() {
        const button = document.getElementById('inscriptionButton');
        const links = [
            'https://www.havan.com.br/',
            'https://www.havan.com.br/',
            'https://www.havan.com.br/',
            'https://www.havan.com.br/'
        ];
        
        // Get the current index from localStorage or set it to 0 if it doesn't exist
        let currentIndex = parseInt(localStorage.getItem('currentLinkIndex')) || 0;

        // Set the button's href to the current link
        button.href = links[currentIndex];

        // Update the index for the next user
        currentIndex = (currentIndex + 1) % links.length;
        localStorage.setItem('currentLinkIndex', currentIndex);
    }

    balanceTraffic();
